CREATE TABLE vinod.Posts
(
    PostId SERIAL PRIMARY KEY,
    UserId INT,
    Title VARCHAR(200),
    Description TEXT,
    UserName VARCHAR(100),
    IsAnonymous BOOLEAN DEFAULT FALSE,
    Status VARCHAR(20) DEFAULT 'Pending',
    ApprovedBy INT,
    RejectedBy INT,
    RejectedReason VARCHAR(500),
    IsPinned BOOLEAN DEFAULT FALSE,
    CreatedDate TIMESTAMP DEFAULT now(),

    CONSTRAINT FK_Posts_Users
    FOREIGN KEY (UserId) REFERENCES vinod.Users(UserId),

    CONSTRAINT FK_Posts_ApprovedBy
    FOREIGN KEY (ApprovedBy) REFERENCES vinod.Users(UserId),

    CONSTRAINT FK_Posts_RejectedBy
    FOREIGN KEY (RejectedBy) REFERENCES vinod.Users(UserId),

    CONSTRAINT CHK_Posts_Status
    CHECK (Status IN ('Pending', 'Approved', 'Rejected'))
);

Select * from vinod.Posts

CREATE TABLE vinod.PostFiles
(
    FileId SERIAL PRIMARY KEY,
    PostId INT,
    FileName VARCHAR(255),
    FileType VARCHAR(50),
    FilePath VARCHAR(500),
    FileUrl VARCHAR(500),
    UploadedDate TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT FK_PostFiles_Posts
    FOREIGN KEY (PostId) REFERENCES vinod.Posts(PostId)
);

CREATE TABLE vinod.ChatMessages
(
    MessageId SERIAL PRIMARY KEY,

    SenderUserId INT NOT NULL,

    ReceiverUserId INT NOT NULL,

    Message TEXT NOT NULL,

    SentDate TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    IsRead BOOLEAN DEFAULT FALSE,

    CONSTRAINT FK_Chat_Sender
    FOREIGN KEY (SenderUserId)
    REFERENCES vinod.Users(UserId),

    CONSTRAINT FK_Chat_Receiver
    FOREIGN KEY (ReceiverUserId)
    REFERENCES vinod.Users(UserId)
);
